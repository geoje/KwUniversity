function even1(num) {
    let even = 2, result = 0;
    while (even <= num) {
        result += even;
        even += 2;
    }
    return result;
}
function even2(num) {
    let even = 2, result = 0;
    do {
        result += even;
        even += 2;
    } while (even <= num);
    return result;
}
function even3(num) {
    let result = 0;
    for (let even = 2; even <= num; even += 2)
        result += even;
    return result;
}
